```python
def _call_provider_with_fallback(
    provider_name: str,
    make_request_fn: Callable[[str], Any],
    api_keys: list[str],
    key_labels: list[str],
) -> Any:
    """Try each provider API key in order, falling back only on HTTP 429 errors.

    Args:
        provider_name: Display name used in logs, e.g. "Mistral" or "Groq".
        make_request_fn: Callable accepting an API key and performing the request.
        api_keys: Ordered key list: primary → secondary → tertiary.
        key_labels: Matching slot labels, never actual key values.

    Returns:
        First successful response.

    Raises:
        LLMConfigurationError if no keys are configured or all keys hit 429.
        RuntimeError immediately for non-429 errors, with key slot context.
    """
    if not api_keys:
        raise LLMConfigurationError(f"No {provider_name} API keys configured.")

    last_error: Exception | None = None
    tried_labels: list[str] = []

    for idx, (key, label) in enumerate(zip(api_keys, key_labels)):
        upper_label = label.upper()
        tried_labels.append(upper_label)

        print(f"[{provider_name}] Attempting request with {upper_label} key", flush=True)
        logger.info("[%s] Attempting request with %s key", provider_name, upper_label)

        try:
            result = make_request_fn(key)
            print(f"[{provider_name}] {upper_label} key succeeded", flush=True)
            logger.info("[%s] %s key succeeded", provider_name, upper_label)
            return result

        except Exception as exc:  # noqa: BLE001
            if _is_rate_limit_error(exc):
                next_labels = key_labels[idx + 1 :]

                if next_labels:
                    next_label = next_labels[0].upper()
                    print(
                        f"[{provider_name}] {upper_label} key hit 429 rate limit. "
                        f"Rotating to {next_label} key...",
                        flush=True,
                    )
                    logger.warning(
                        "[%s] %s key hit 429 rate limit. Rotating to %s key.",
                        provider_name,
                        upper_label,
                        next_label,
                    )
                else:
                    print(
                        f"[{provider_name}] {upper_label} key hit 429 rate limit. "
                        "No more keys to try.",
                        flush=True,
                    )
                    logger.warning(
                        "[%s] %s key hit 429 rate limit. No more keys to try.",
                        provider_name,
                        upper_label,
                    )

                last_error = exc
                continue

            print(
                f"[{provider_name}] {upper_label} key raised a non-429 error "
                f"({type(exc).__name__}); not rotating.",
                flush=True,
            )
            logger.error(
                "[%s] %s key raised non-429 error (%s); propagating immediately.",
                provider_name,
                upper_label,
                type(exc).__name__,
            )

            key_context = f"[Key: {upper_label}]"
            raise RuntimeError(f"{key_context} {type(exc).__name__}: {exc}") from exc

    key_chain = "→".join(tried_labels)
    exhausted_context = f"[All keys exhausted: {key_chain}]"

    print(
        f"[{provider_name}] {exhausted_context} All API keys hit 429 rate limit. "
        "Please wait before retrying or add more API keys.",
        flush=True,
    )
    logger.warning(
        "[%s] %s All API keys hit 429 rate limit.",
        provider_name,
        exhausted_context,
    )

    raise LLMConfigurationError(
        f"{exhausted_context} All {provider_name} API keys exhausted due to "
        "rate limiting (429). Please wait before retrying or add more API keys."
    ) from last_error
```

def _get_provider_key_chain(provider: str) -> tuple[list[str], list[str]]:
    """Return ordered API key chain and labels for fallback-capable providers."""
    provider_upper = provider.upper()

    if provider == "mistral":
        keys = [
            _env("MISTRAL_API_KEY_PRIMARY") or _env("MISTRAL_API_KEY"),
            _env("MISTRAL_API_KEY_SECONDARY"),
            _env("MISTRAL_API_KEY_TERTIARY"),

        keys = [
            _env("GROQ_API_KEY_PRIMARY") or _env("GROQ_API_KEY"),
            _env("GROQ_API_KEY_SECONDARY"),
            _env("GROQ_API_KEY_TERTIARY"),
        ]
    else:
        return [], []

    labels = ["primary", "secondary", "tertiary"]

    filtered_keys: list[str] = []
    filtered_labels: list[str] = []

    for key, label in zip(keys, labels):
        if key:
            filtered_keys.append(key)
            filtered_labels.append(label)

    if not filtered_keys:
        raise LLMConfigurationError(f"No {provider_upper} API keys configured.")

    return filtered_keys, filtered_labels
